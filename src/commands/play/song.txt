    }

// Initialize cookies once
let cookiesInitialized = false;
let soundcloudInitialized = false;

export const config = createCommandConfig({
    description: 'Play music from YouTube, SoundCloud, or direct audio URLs.',
    options: [
        {
            name: 'song',
            description: 'Song name, YouTube URL, SoundCloud URL, or direct audio URL',
            type: 'string',
            required: true
        }
    ]
} as const);

export default async (interaction: ChatInputCommandInteraction) => {
    await interaction.deferReply({ ephemeral: true });

    const optionSong = interaction.options.getString('song');

    if (!optionSong) {
        return interaction.editReply({
            content: 'Please provide a valid song name.'
        });
    }

    if(interaction.guild === null) {
        return interaction.editReply({
            content: 'This command can only be used in a server.'
        });
    }

    const guild = client.guilds.cache.get(interaction.guild.id);
    const member = guild?.members.cache.get(interaction.user.id);

    if(!guild || !member) {
        return interaction.editReply({
            content: 'There was an error retrieving the guild or member information.'
        });
    }

    const voiceChannel = member.voice.channel?.id;

    if(!voiceChannel) {
        return interaction.editReply({
            content: 'You need to be in a voice channel to play a song.'
        });
    }

    // Initialize play-dl with cookies and SoundCloud auth if needed
    if (!cookiesInitialized) {
        await interaction.editReply({
            content: 'ðŸ”§ Setting up YouTube authentication...'
        });
        await setupYouTubeCookies();
        cookiesInitialized = true;
    }

    if (!soundcloudInitialized) {
        await interaction.editReply({
            content: 'ðŸ”§ Setting up SoundCloud authentication...'
        });
        await setupSoundCloudAuth();
        soundcloudInitialized = true;
    }

    let song;
    let songName = optionSong; // Default to the search term

    // Check if it's a URL and determine platform
    const isYouTubeUrl = optionSong.includes('youtube.com') || optionSong.includes('youtu.be');
    const isSoundCloudUrl = optionSong.includes('soundcloud.com');
    const isDirectUrl = optionSong.startsWith('http') && !isYouTubeUrl && !isSoundCloudUrl;
    const isUrl = isYouTubeUrl || isSoundCloudUrl || isDirectUrl;

    try {
        if (isUrl) {
            if (isYouTubeUrl) {
                await interaction.editReply({
                    content: 'ðŸ”— Processing YouTube URL...'
                });
                song = optionSong;
                console.log(`ðŸ”— Using YouTube URL: ${song}`);
                
                try {
                    const info = await play.video_basic_info(optionSong);
                    songName = info.video_details.title || songName;
                    console.log(`ðŸ“ Song title: ${songName}`);
                } catch {
                    songName = 'YouTube Song';
                    console.log(`âš ï¸ Could not get YouTube video info, using fallback name`);
                }
            } else if (isSoundCloudUrl) {
                await interaction.editReply({
                    content: 'ðŸŽµ Processing SoundCloud URL...'
                });
                song = optionSong;
                console.log(`ðŸ”— Using SoundCloud URL: ${song}`);
                
                // For SoundCloud URLs, we'll get the title during streaming
                // to avoid the authorization error here
                songName = 'SoundCloud Track';
                console.log(`ðŸ“ SoundCloud URL processed, will get track info during streaming`);
            } else if (isDirectUrl) {
                await interaction.editReply({
                    content: 'ðŸ”— Processing direct audio URL...'
                });
                song = optionSong;
                songName = 'Direct Audio Stream';
                console.log(`ï¿½ Using direct URL: ${song}`);
            }
        } else {
            await interaction.editReply({
                content: `ï¿½ðŸ” Searching for: **${optionSong}**...`
            });
            console.log(`ðŸ” Searching for: ${optionSong}`);
            
            // Try YouTube first, then SoundCloud as fallback
            let searchResults = [];
            let searchPlatform = 'unknown';
            
            try {
                // First try YouTube search
                console.log('ðŸ” Trying YouTube search...');
                searchResults = await play.search(optionSong, { 
                    limit: 1,
                    source: { youtube: "video" }
                });
                searchPlatform = 'YouTube';
            } catch (searchError) {
                console.log('âŒ YouTube search failed, trying SoundCloud...');
                await interaction.editReply({
                    content: `ðŸ” YouTube search failed, trying SoundCloud for: **${optionSong}**...`
                });
                
                try {
                    searchResults = await play.search(optionSong, { 
                        limit: 1,
                        source: { soundcloud: "tracks" }
                    });
                    searchPlatform = 'SoundCloud';
                } catch (soundcloudError) {
                    console.log('âŒ SoundCloud search failed, trying general search...');
                    await interaction.editReply({
                        content: `ðŸ” Trying general search for: **${optionSong}**...`
                    });
                    
                    try {
                        searchResults = await play.search(optionSong, { limit: 1 });
                        searchPlatform = 'General';
                    } catch (fallbackError) {
                        console.error('âŒ All search methods failed:', fallbackError);
                        return interaction.editReply({
                            content: 'Search failed on all platforms. Please try with a direct URL (YouTube, SoundCloud, or direct audio link).'
                        });
                    }
                }
            }
            
            console.log(`ðŸ“Š Search results from ${searchPlatform}:`, searchResults.length);
            
            if (searchResults.length === 0) {
                return interaction.editReply({
                    content: 'No results found on any platform. Try using a direct URL or different search terms.'
                });
            }
            
            songName = searchResults[0].title || searchResults[0].name || 'Unknown Song';
            song = searchResults[0].url;
            console.log(`âœ… Found song on ${searchPlatform}: ${songName}`);
            console.log(`ðŸ”— Song URL: ${song}`);
            
            await interaction.editReply({
                content: `âœ… Found on ${searchPlatform}: **${songName}**`
            });
        }
        
        // Validate that we have a song URL
        if (!song) {
            console.error('âŒ Song URL is undefined');
            return interaction.editReply({
                content: 'Failed to get song URL. Please try again.'
            });
        }
        
    } catch (error) {
        console.error('Error validating/searching song:', error);
        return interaction.editReply({
            content: 'Failed to find or validate the song. Try using a direct YouTube URL or check if the song exists.'
        });
    }

    try {
        console.log(`ðŸŽµ Attempting to stream: ${song}`);
        
        await interaction.editReply({
            content: `ðŸ“‹ Validating video: **${songName}**...`
        });
        
        // Try to get video/track info first to validate the URL and determine platform
        let videoInfo;
        let platform = 'Unknown';
        
        try {
            if (song.includes('soundcloud.com')) {
                console.log('ðŸŽµ Detected SoundCloud URL, attempting direct streaming...');
                platform = 'SoundCloud';
                // Skip getting track info to avoid authorization error
                // We'll get the info during streaming
                console.log(`ðŸ“‹ SoundCloud URL detected, proceeding to stream`);
            } else if (song.includes('youtube.com') || song.includes('youtu.be')) {
                console.log('ðŸ“º Detected YouTube URL, getting video info...');
                videoInfo = await play.video_basic_info(song);
                platform = 'YouTube';
                console.log(`ðŸ“‹ YouTube video info retrieved: ${videoInfo.video_details.title}`);
            } else {
                console.log('ðŸ”— Direct URL detected, attempting to stream...');
                platform = 'Direct URL';
            }
        } catch (infoError) {
            console.error('âŒ Failed to get media info:', infoError);
            // For SoundCloud, we'll continue and try streaming directly
            if (song.includes('soundcloud.com')) {
                console.log('âš ï¸ SoundCloud info failed, but continuing with direct streaming...');
                platform = 'SoundCloud';
            } else {
                return interaction.editReply({
                    content: `Failed to access ${platform} information. The media might be private, region-locked, or unavailable.`
                });
            }
        }
        
        await interaction.editReply({
            content: `ðŸŽµ Preparing ${platform} stream for: **${songName}**...`
        });
        
        // Try streaming with multiple methods based on platform
        let stream: any;
        
        try {
            console.log(`ðŸŽµ Method 1: Platform-specific streaming for ${platform}...`);
            
            await interaction.editReply({
                content: `ðŸŽµ Creating ${platform} stream for: **${songName}**...`
            });
            
            if (platform === 'SoundCloud') {
                // SoundCloud streaming - use direct URL approach
                console.log('ðŸŽµ Using SoundCloud direct stream...');
                stream = await play.stream(song, {
                    discordPlayerCompatibility: true,
                    quality: 1
                });
            } else if (platform === 'YouTube') {
                // YouTube streaming with fresh info
                const freshVideoInfo = await play.video_basic_info(song);
                console.log('ðŸ“‹ Fresh YouTube info obtained, streaming immediately...');
                stream = await play.stream_from_info(freshVideoInfo, {
                    discordPlayerCompatibility: true,
                    quality: 1
                });
            } else {
                // Direct URL streaming
                console.log('ðŸ”— Using direct URL stream...');
                stream = await play.stream(song, {
                    discordPlayerCompatibility: true
                });
            }
            
            console.log(`âœ… Method 1 succeeded - ${platform} stream`);
            
        } catch (error: any) {
            console.log(`âŒ Method 1 failed for ${platform}:`, error.message);
            
            // Fallback methods
            try {
                console.log('ðŸ”„ Method 2: Using direct URL with compatibility settings...');
                await interaction.editReply({
                    content: `ðŸ”„ Trying alternative method for: **${songName}**...`
                });
                
                stream = await play.stream(song, { 
                    discordPlayerCompatibility: true,
                    quality: 0 // Lowest quality for maximum compatibility
                });
                console.log('âœ… Method 2 succeeded - direct URL low quality');
                
            } catch (compatError: any) {
                console.log('âŒ Method 2 failed:', compatError.message);
                
                try {
                    console.log('ðŸ”„ Method 3: Basic stream without options...');
                    await interaction.editReply({
                        content: `ðŸ”„ Last attempt for: **${songName}**...`
                    });
                    
                    stream = await play.stream(song);
                    console.log('âœ… Method 3 succeeded - basic stream');
                    
                } catch (basicError: any) {
                    console.log('âŒ Method 3 failed:', basicError.message);
                    
                    // Final fallback with delay
                    try {
                        console.log('ðŸ”„ Method 4: Delayed retry...');
                        await interaction.editReply({
                            content: `ðŸ”„ Final attempt for: **${songName}**...`
                        });
                        
                        // Small delay to let any rate limiting reset
                        await new Promise(resolve => setTimeout(resolve, 2000));
                        
                        if (platform === 'SoundCloud') {
                            stream = await play.stream(song);
                        } else if (platform === 'YouTube') {
                            const retryVideoInfo = await play.video_basic_info(song);
                            stream = await play.stream_from_info(retryVideoInfo);
                        } else {
                            stream = await play.stream(song);
                        }
                        console.log('âœ… Method 4 succeeded - delayed retry');
                        
                    } catch (finalError: any) {
                        console.error('âŒ All streaming methods failed:', {
                            platform: platform,
                            method1: error.message,
                            method2: compatError.message,
                            method3: basicError.message,
                            method4: finalError.message,
                            songUrl: song
                        });
                        
                        return interaction.editReply({
                            content: `âŒ **Unable to stream from ${platform}: ${songName}**\n\`\`\`\nAll streaming methods failed.\nThis ${platform.toLowerCase()} track may be:\nâ€¢ Region-locked\nâ€¢ Age-restricted\nâ€¢ Blocked for bots\nâ€¢ Recently made private\n\`\`\`\nðŸ’¡ **Try:** A different track, platform, or direct audio URL`
                        });
                    }
                }
            }
        }
        
        if (!stream) {
            console.error('âŒ Stream is null or invalid');
            return interaction.editReply({
                content: 'Failed to create audio stream. The song might be unavailable.'
            });
        }
        
        await interaction.editReply({
            content: `ðŸ”— Connecting to voice channel for: **${songName}**...`
        });
        
        const player = createAudioPlayer();
        let resource;
        
        // Create audio resource from the stream
        try {
            if (stream.stream) {
                // play-dl stream object
                resource = createAudioResource(stream.stream, {
                    inputType: stream.type
                });
                console.log('ðŸ“¦ Created resource from play-dl stream object');
            } else {
                // Direct stream
                resource = createAudioResource(stream);
                console.log('ðŸ“¦ Created resource from direct stream');
            }
        } catch (resourceError: any) {
            console.error('âŒ Failed to create audio resource:', resourceError);
            return interaction.editReply({
                content: `âŒ Failed to create audio resource: ${resourceError.message}`
            });
        }
        const connection = joinVoiceChannel({
            channelId: voiceChannel,
            guildId: guild.id,
            adapterCreator: guild.voiceAdapterCreator
        });

        connection.subscribe(player);
        player.play(resource);

        player.on('error', error => {
            console.error('Error playing song:', error);
            interaction.editReply({
                content: `âŒ An error occurred while playing: **${songName}**`
            });
        });

        player.on(AudioPlayerStatus.Idle, () => {
            connection.destroy();
            interaction.editReply({
                content: `âœ… Finished playing: **${songName}**`
            });
        });

        await interaction.editReply({
            content: `ðŸŽ¶ Now playing: **${songName}**`
        });
        
    } catch (error) {
        console.error('Error streaming song:', error);
        return interaction.editReply({
            content: 'Failed to stream the song. Please try again.'
        });
    }
};
